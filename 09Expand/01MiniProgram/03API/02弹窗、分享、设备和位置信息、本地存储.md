### 1.弹窗

```html
<button bindtap="onShowToast">showToast</button>
<button bindtap="onShowModal">showModal</button>
<button bindtap="onShowAction">showAction</button>
```

```js
Page({
  onShowToast() {
    wx.showToast({
      title: '购买成功',
      icon: "success",
      duration: 2000,
      success: res => console.log(res),
      fail: err => console.log(err)
    })
  },
  onShowModal() {
    wx.showModal({
      title: "确定购买吗?",
      content: "请确保微信零钱有足够的金额",
      confirmColor: "#0f0",
      cancelColor: '#f00',
      success: res => {
        if(res.cancel) {
          console.log("用户点击了取消");
        } else if (res.confirm) {
          console.log("用户点击了确定");
        }
      }
    })
  },
  onShowAction() {
    wx.showActionSheet({
      itemList: ["电脑", "手机", "平板"],
      success: res => console.log(res.tapIndex),
      fali: err => console.log(err)
    })
  }
})
```

### 5.本地存储

```html
<button bindtap="onLocalStorage">本地存储</button>
```

```js
onLocalStorage() {
  wx.setStorageSync('name', "yt")
  wx.setStorageSync('age', 22)
  wx.setStorageSync('hobbies', ["游戏", "篮球", "读书"])

  const name = wx.getStorageSync('name')
  const age = wx.getStorageSync('age')
  const hobbies = wx.getStorageSync('hobbies')
  console.log(name, age, hobbies);
}
```

```js
// 异步
onLocalStorage() {
  wx.setStorage({
    key: "books",
    data: "《明朝那些事儿》",
    encrypt: true,  // 加密, 注意版本比较高
    success: () => {
      wx.getStorage({
        key: "books",
        encrypt: true,
        success: res => console.log(res)
      })
    }
  })
}
```
