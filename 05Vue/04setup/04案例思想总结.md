### 1.异步获取json文件中数据

```js
import { ref } from "vue"

const headerTitle = ref({title: "", subTitle: ""})
const contentList = ref([])

setTimeout(async () => {
  const res = await import("../assets/data/high_score.json")
  headerTitle.value.title = res.default.title
  headerTitle.value.subTitle = res.default.subtitle
  contentList.value = res.default.list
}, 600)
```

### 2.计算属性处理复杂逻辑

```html
<script setup>
import { computed } from "vue"

const props = defineProps({
  item: {
    type: Object,
    default: () => ({})
  }
})

const textContent = computed(() => ({
  text: props.item.verify_info.messages.join(" "),
  textColor: props.item.verify_info.text_color
}))

const bottomInfo = computed(() => ({
  content: props.item.bottom_info.content,
  color: props.item.bottom_info.content_color
}))
</script>

<template>
  <div class="text" :style="{color: textContent.textColor}">{{textContent.text}}</div>
  <div :style="{color: bottomInfo.color}">{{bottomInfo.content}}</div>
</template>
```
