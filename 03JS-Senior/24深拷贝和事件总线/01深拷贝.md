```html
<script>
  // 判断一个变量是否是对象类型
  function isObject(value) {
    const valueType = typeof value
    return (value !== null) && (valueType === "object" || valueType === "function")
  }

  function deepCopy(obj) {
    if (!isObject(obj)) {
      return obj
    }

    const newObj = {}
    for (const key in obj) {
      newObj[key] = deepCopy(obj[key])
    }
    return newObj
  }

  const infoObj = {
    name: "yt",
    age: 22,
    friend: {
      name: "张三",
      age: 22,
      hobbies: {
        name: "王者荣耀",
        time: "6年"
      }
    }
  }

  const newInfoObj = deepCopy(infoObj);
</script>
```
