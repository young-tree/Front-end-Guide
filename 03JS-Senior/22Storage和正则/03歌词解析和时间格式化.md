const lyricString = "[00:00.000] 作词 : 许嵩\n[00:01.000] 作曲 : 许嵩\n[00:02.000] 编曲 : 许嵩\n[00:22.240]天空好想下雨\n[00:24.38]我好想住你隔壁\n[00:26.810]傻站在你家楼下\n[00:29.500]抬起头数乌云\n[00:31.160]如果场景里出现一架钢琴\n"
    
const lyricArry = lyricString.split("\n")
lyricArry.length = lyricArry.length - 1

const re = /\[(\d{2}):(\d{2})\.(\d{2,3})\]/i
finalLyricArry = []
for (const lyric of lyricArry) {
  const timeInfo = lyric.match(re);
  const minute = timeInfo[1] * 60 *1000
  const second = timeInfo[2] * 1000
  const mSecond = timeInfo[3].padEnd(3, "0") * 1
  const totalTime = minute + second + mSecond

  const lyricInfo = lyric.replace(re, "").trim()

  finalLyricArry.push({totalTime, lyricInfo})
}

console.log(finalLyricArry)

function timeFormater(timeStamp, string) {
  const time = new Date(timeStamp)
  const timeRegObj = {
    "y+": time.getFullYear(),
    "M+": time.getMonth() + 1,
    "d+": time.getDate(),
    "h+": time.getHours(),
    "m+": time.getMinutes(),
    "s+": time.getSeconds()
  }

  for (const key in timeRegObj) {
    const reg = new RegExp(key)
    const value = (timeRegObj[key] + "").padStart(2, "0")
    string = string.replace(reg, value)
  }

  return string
}

const timeStamp = 1659252290626
const timeString = timeFormater(timeStamp, "yyyy/MM/dd hh:mm:ss")
console.log(timeString)
