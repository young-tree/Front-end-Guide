### 1.length属性

```js
// 1.获取数组的长度
console.log(names.length);

// 2.length属性是可写的, 可以用于扩容, 但是多出来的是空元素
names.length = 10;
console.log(names);

// 3.也可以用于删除部分元素或者清空数组
names.length = 2;
console.log(names);
names.length = 0;
console.log(names);
```

### 2.遍历数组

```js
var stu = ["msi", "lol", "lpl", "ldl"]

// 1.普通for循环
for(var i = 0; i < stu.length; i++) {
  console.log(stu[i]);
}

// 2.for in
for(var index in stu) {
  console.log(stu[index]);
}

// 3.for of
for(var item of stu) {
  console.log(item);
}
```

### 3.slice-concat-join

```js
var stu = ["msi", "lol", "lpl", "ldl"];

// 1.slice方法: 不会修改原数组
console.log(stu.slice(2, 3));

// 2.concat方法: 将多个数组拼接在一起
var name1 = ["abc", "cba"];
var name2 = ["msi", "lol"];
var name3 = ["lpl", "nba"];

console.log(name1.concat(name2, name3));
// concat是一个纯函数, 不会改变原数组, 而是返回一个新的数组

// 3.join方法: 将数组的元素以什么分割符拼接在一起, 组成字符串
console.log(stu.join("%"));
```

### 4.查找元素

- 查找数组中的某个原始数据类型

  ```js
  var stu = ["msi", "lol", "lpl", "ldl"];
  
  // 1.indexOf
  console.log(stu.indexOf("lol"));  // 1
  console.log(stu.indexOf("ms"));  // -1
  // 2.lastIndexOf
  // 3.includes
  ```

- 查找数组中的某个对象

  ```js
  var students = [
    { id: 100, name: "why", age: 18 },
    { id: 101, name: "kobe", age: 30 },
    { id: 102, name: "james", age: 25 },
    { id: 103, name: "curry", age: 22 }
  ]
  // 查找students数组中id为101的学生信息, 这里就不能用indexOf或者includes了
  
  // 1. 通过find方法实现, 直接返回查找到的当前元素
  console.log(students.find(item => item.id === 101));
  // 2. findIndex, 返回查找到的当前元素的索引值
  ```

### 5.手动实现forEach和find方法

```js
var stu = ["msi", "lol", "lpl", "ldl"];

// 1.手动实现forEach方法
Array.prototype.YTForEach = function(fn, thisArg) {
  for (var i = 0; i < this.length; i++) {
    fn.apply(thisArg, [this[i], i, this]);
  }
}

stu.YTForEach(item => console.log(item, this), 123);
// 这里本来是可以绑定123的, 但是使用了箭头函数, 所以this指向window
    
// 2.手动实现find方法
var students = [
  { id: 100, name: "why", age: 18 },
  { id: 101, name: "kobe", age: 30 },
  { id: 102, name: "james", age: 25 },
  { id: 103, name: "curry", age: 22 }
]

Array.prototype.YTFind = function(fn, thisArg) {
  for (var i = 0; i < this.length; i++) {
    if (fn.apply(thisArg, [this[i], i, this])) {
      return this[i];
    }
  }
}

console.log(students.YTFind(item => item.id === 101));
```

### 6.其他高阶函数的使用

```js
var stu = ["msi", "lol", "lpl", "ldl"];

// 1. forEach方法
stu.forEach(item => console.log(item, this));

// 2. filter方法
var nums = [1, 2, 3, 4, 5, 6, 7, 8];
// 2.1. 通过for循环实现取出偶数
var newNums = []
for (var i = 0; i < nums.length; i++) {
  if (nums[i] % 2 === 0) {
    newNums.push(nums[i]);
  }
}
console.log(newNums);
// 2.2. 通过filter方法实现取出偶数
console.log(nums.filter(item => item % 2 === 0));

// 3. map方法
// 每一个元素都平方
console.log(nums.map(item => item ** 2));

// 4. reduce方法
// 累加
console.log(nums.reduce((preVal, item) => preVal + item, 0))

// 5.reduce方法的练习
var products = [
  { name: "鼠标", price: 88, count: 3 },
  { name: "键盘", price: 200, count: 2 },
  { name: "耳机", price: 9.9, count: 10 },
]
// 计算总价
console.log(products.reduce((preVal, item) => preVal + item.price * item.count, 0));

// 6.综合练习
var nums = [11, 20, 55, 100, 88, 32]
// 过滤所有的偶数, 映射所有偶数的平方, 并且计算他们的和
var abc = nums.filter(item => item % 2 === 0).map(item => item ** 2)
              .reduce((preVal, item) => preVal + item, 0);
console.log(abc)

var cba = nums.filter(item => item % 2 === 0)
              .reduce((preVal, item) => preVal + item ** 2, 0);
console.log(cba);
```

### 7.排序和反转

- 排序：可以改变原数据，不是纯函数

  ```js
  // 1.原始数据类型的排序
  var nums = [20, 4, 10, 15, 100, 88]
  
  console.log(nums.sort((item1, item2) => item1 - item2));  // 升序
  console.log(nums.sort((item1, item2) => item2 - item1));  // 降序
  
  // 2.复杂数据类型的排序
  var students = [
    { id: 100, name: "why", age: 18 },
    { id: 101, name: "kobe", age: 30 },
    { id: 102, name: "james", age: 25 },
    { id: 103, name: "curry", age: 22 }
  ]
  // 以年龄的大小进行排序
  console.log(students.sort((item1, item2) => item1.age - item2.age));  // 升序
  console.log(students.sort((item1, item2) => item2.age - item1.age));  // 降序
  ```

- 反转：可以改变原数据，也不是纯函数（既可以改变原数组，又可以生成新的数组）

  ```js
  var nums = [20, 4, 10, 15, 100, 88]
  
  console.log(nums.reverse());
  ```

- 这两个确实不是纯函数，它具体的操作应该是，把原数组经过排序，改变成升序或者降序数组，再将其返回，返回的数组和原数组其实是一个数组，并不是重新生成一个新的数组

  ```js
  var nums = [20, 4, 10, 15, 100, 88]
  
  console.log(nums.sort((item1, item2) => item1 - item2) === nums);  // true
  ```
