```html
<script>
  // 1.setTimeout
  const id1 = setTimeout(() => console.log("setTimeout"), 1000);

  setTimeout((name, age, height) => {
    clearTimeout(id1);
    console.log(name, age, height);
  }, 500, "yt", 22, 1.88);

  // 2.setInterval
  const id2 = setInterval(() => console.log("setInterval"), 1000);

  setTimeout(() => clearInterval(id2), 500);

  // 3.清除计时器后如何重启计时器
  const fn = () => console.log("setInterval");

  const id3 = setInterval(fn, 1000);

  setTimeout(() => {
    clearInterval(id3);
    setInterval(fn, 1000);
  }, 500);
</script>
```
