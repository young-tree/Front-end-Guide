### 1.鼠标事件

```html
<div class="box">box</div>
<script>
  const boxEl = document.querySelector(".box");

  // 1.点击事件
  boxEl.addEventListener("click", () => {
    console.log("click");
  });

  // 2.右键点击事件
  boxEl.addEventListener("contextmenu", (event) => {
    event.preventDefault();
    console.log("contextmenu");
  });

  // 3.按下 移动 抬起事件
  boxEl.addEventListener("mousedown", () => {
    console.log("mousedown")
  });
  boxEl.addEventListener("mousemove", () => {
    console.log("mousemove");
  });
  boxEl.addEventListener("mouseup", () => {
    console.log("mouseup");
  });

  // 关于mousemove, 只要移上去就会打印
  // 现在有一个需求: 按下鼠标之后才会打印mousemove, 鼠标抬起停止打印mousemove
  let isDown = false;
  boxEl.addEventListener("mousedown", () => {
    isDown = true;
  });
  boxEl.addEventListener("mousemove", () => {
    if (!isDown) return;
    console.log("mousemove");
  });
  boxEl.addEventListener("mouseup", () => {
    isDown = false;
  });
</script>
```

### 2.enter和over

```html
<div class="wrap">
  <div class="box"></div>
</div>

<script>
  const wrapEl = document.querySelector(".wrap");
  const boxEl = document.querySelector(".box");

  wrapEl.addEventListener("mouseenter", () => console.log("外层"));
  boxEl.addEventListener("mouseenter", () => console.log("内层"));

  // 外层比内层大
  // 进入外层时打印外层, 进入内层时打印内层, 没有冒泡
  // 外层 内层

  // 外层和内层一样大
  // 由于一样大, 所以进入内层实际上是先进入外层, 所以会同时打印外层和内层, 先外层后内层
  // 外层 内层

  // 外层比内层小
  // 虽然外层比内层小, 但是依然同时打印内层和外层, 先打印外层后打印内层
  // 外层 内层


  wrapEl.addEventListener("mouseover", () => console.log("外层"));
  boxEl.addEventListener("mouseover", () => console.log("内层"));

  // 外层比内层大
  // 进入外层打印外层, 进入内层冒泡, 其实进入内层的同时外层的mouseover就结束了
  // 所以进入内层会先打印内层再打印外层
  // 离开内层进入外层, 再打印外层
  // 外层 内层 外层 外层

  // 外层和内层一样大
  // 直接进入内层, 然后冒泡, 先打印内层后打印外层
  // 也就是说不会像mouseenter一样, 先经过外层再经过内层
  // 内层 外层

  // 外层比内层小
  // 效果和内外层一样大一样
  // 内层 外层

  wrapEl.addEventListener("mouseleave", () => console.log("外层"));
  boxEl.addEventListener("mouseleave", () => console.log("内层"));
  // 先离开内层再离开外层 打印: 
  // 内层 外层

  wrapEl.addEventListener("mouseout", () => console.log("外层"));
  boxEl.addEventListener("mouseout", () => console.log("内层"));
  // 一样大和外层比内层小, 均是离开内层时打印: 
  // 内层 外层

  // 外层比内层大时
  // 当我们进入内层时, 离开外层, 打印外层
  // 当我们离开内层时, 打印内层 由于冒泡 所以也会打印外层
  // 离开外层时 打印外层
  // 外层 内层 外层 外层


  // mouseenter必须经历外层
  // mouseover内层被完全覆盖不经历外层
</script>
```

### 3.键盘事件

```html
<input type="text">
<button>搜索</button>

<script>
  const inputEl = document.querySelector("input");
  const buttonEl = document.querySelector("button");

  // 1.按键的事件
  inputEl.addEventListener("keydown", () => console.log("keydown"));  // 键按下和一直按着
  inputEl.addEventListener("keypress", () => console.log("keypress"));  // 键一直按着
  // 感觉它俩一样

  inputEl.addEventListener("keyup", () => console.log("keyup"));  // 键抬起

  // 2.按钮搜索和回车搜索
  buttonEl.addEventListener("click", () => {
    console.log("按钮搜索: ", inputEl.value);
  });

  inputEl.addEventListener("keyup", (event) => {
    if (event.code === "Enter") {
      console.log("回车搜索: ", inputEl.value);
    }
  });

  // 3.按s后input框获取焦点
  window.addEventListener("keyup", (event) => {
    if (event.code === "KeyS") {
      inputEl.focus();
    }
  });
  // 除了code属性还有一个key属性
</script>
```

### 4.input事件

```html
<form action="/abc">
  <input type="text">

  <button type="reset">重置</button>
  <button type="submit">提交</button>
</form>
<script>
  const inputEl = document.querySelector("input");

  // 1.获取焦点和失去焦点
  inputEl.addEventListener("focus", () => {
    console.log("inputEl获得焦点~focus");
  });
  inputEl.addEventListener("blur", () => {
    console.log("inputEl失去焦点~blur");
  });

  // 2.input事件和change事件
  // input事件: 只要输入内容就会触发
  inputEl.addEventListener("input", () => {
    console.log("input事件~");
  });
  // change事件: 输入完毕后, 焦点失去, 确定是这个内容就会触发
  // 不是特别好用
  inputEl.addEventListener("change", () => {
    console.log("change事件~");
  });

  // 3.重置事件和提交事件
  // 首先我们要明白, 监听重置和提交, 监听的是form表单
  const formEl = document.querySelector("form");
  formEl.addEventListener("reset", () => {
    console.log("重置事件~");
  });

  formEl.addEventListener("submit", (event) => {
    event.preventDefault();
    console.log("提交事件~");
  });
</script>
```

### 5.文档事件

```html
<script>
  // 1.文档加载完毕, 资源并没有加载完毕
  window.addEventListener("DOMContentLoaded", () => {
    const boxEl = document.querySelector(".box");
    boxEl.classList.add("change");

    const imgEl = document.querySelector("img");
    console.log(imgEl.offsetWidth);
    console.log(imgEl.offsetHeight);
  });

  // 2.文档加载完毕, 资源也加载完毕才会回调
  window.addEventListener("load", () => {
    const boxEl = document.querySelector(".box");
    boxEl.classList.add("change");

    const imgEl = document.querySelector("img");
    console.log(imgEl.offsetWidth);
    console.log(imgEl.offsetHeight);
  })

  // 1.监听屏幕尺寸变化
  window.addEventListener("resize", () => {
    console.log(window.innerWidth);
    console.log(window.innerHeight);
  })
</script>
```
