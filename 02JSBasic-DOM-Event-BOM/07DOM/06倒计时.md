### 倒计时

```html
<div class="countdown">
  <span class="time hours">12</span>
  <span class="division">:</span>
  <span class="time minutes">12</span>
  <span class="division">:</span>
  <span class="time seconds">12</span>
</div>
<script src="./js/formatTime.js"></script>
<script>
  function getCountdown() {
    // 获取当前时间戳
    const nowTime = new Date();
    const nowTimeStamp = nowTime.getTime();
    console.log(nowTimeStamp);

    // 获取结束的时间戳
    nowTime.setHours(24);
    nowTime.setMinutes(0);
    nowTime.setSeconds(0);
    const endTimeStamp = nowTime.getTime();

    // 获取两者时间戳之差
    const remainTimeStamp = endTimeStamp - nowTimeStamp;
    // 获取剩余秒钟
    const remainTimeSeconds = remainTimeStamp / 1000  // 获取的是毫秒, 一定要除以1000获取秒钟

    // 将两者之差转成: 时分秒
    // 在转成时分秒之前, 思考一个问题, 如何将一个数字转成百十个
    // 比如123
    // 百: Math.floor(123 / (10 * 10))
    // 十: Math.floor(123 / 10) % 10
    // 个: 123 % 10

    // 类比于时间
    // 小时: Math.floor(总秒数 / (60 * 60))
    // 分钟: Math.floor(总秒数 / 60) % 60
    // 秒钟: 总秒数 % 60

    const hours = formatTime(Math.floor(remainTimeSeconds / (60 * 60)));
    const minutes = formatTime(Math.floor(remainTimeSeconds / 60) % 60);
    const seconds = formatTime(remainTimeSeconds % 60);

    const hoursEl = document.querySelector(".hours");
    hoursEl.textContent = hours;
    const minutesEl = document.querySelector(".minutes");
    minutesEl.textContent = minutes;
    const secondsEl = document.querySelector(".seconds");
    secondsEl.textContent = seconds;
  }
  getCountdown();

  setInterval(() => {
    getCountdown();
  }, 1000);
</script>
```
