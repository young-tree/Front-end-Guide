### 1.动态的生成列表

```html
<ul></ul>
<script>
  const ulEl = document.querySelector("ul");
  let noStop = true;
  while(noStop) {
    const result = prompt("请输入~");
    if (result) {
      const liEl = document.createElement("li");
      liEl.textContent = result;
      ulEl.append(liEl);
    } else {
      noStop = false;
    }
  }
</script>
```

### 2.动态的显示时间

```js
const formatTime = (time, count = 2, stay = "0") => (time + "").padStart(count, stay);
```

```html
<div class="nowTime"></div>
<script src="./js/formatTime.js"></script>
<script>
  const nowTimeEl = document.querySelector(".nowTime");

  function timeEnd() {
    const date = new Date();
    const year = date.getFullYear();
    const month = formatTime(date.getMonth() + 1);
    const day = formatTime(date.getDate());
    const hours = formatTime(date.getHours());
    const minutes = formatTime(date.getMinutes());
    const seconds = formatTime(date.getSeconds());

    const timeStr = `${year}-${month}-${day} ${hours}:${minutes}:${seconds}`;
    nowTimeEl.textContent = timeStr;
  }
  timeEnd();
  setInterval(() => {
    timeEnd();
  }, 1000);
</script>
```
